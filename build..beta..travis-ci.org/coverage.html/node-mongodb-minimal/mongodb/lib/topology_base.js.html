Store.prototype.execute = function() {
  // Get current ops
  var ops = this.s.storedOps;
  // Reset the ops
  this.s.storedOps = [];
 
  // Execute all the stored ops
  while(ops.length > 0) {
    var op = ops.shift();
 
    if(op.t == 'cursor') {
      op.o[op.m].apply(op.o, op.p);
    } else {
      this.s.topology[op.t](op.n, op.o, op.op, op.c);
    }
  }
}
