  operationId: 1,

        commandName: commandName,

        connectionId: connection

      self.emit('started', command)

      var startTime = new Date().getTime();

      args.push(function(err, r) {
        var endTime = new Date().getTime();

          operationId: ourOpId,

        if(err) {

        callback(err, r);
